def pos_quat_to_matrix(position, quaternion):
    """
    Convert a position vector and quaternion orientation into a 4x4
    homogeneous transformation matrix.

    This function constructs a transformation matrix using a translation
    vector and a quaternion (scalar-last format). The resulting matrix
    follows the standard homogeneous transformation convention:

        T = | R  t |
            | 0  1 |

    where:
        - R is a 3x3 rotation matrix
        - t is a 3x1 translation vector

    Args:
        position (array-like): A 3-element iterable [x, y, z]
            representing the translation vector.
        quaternion (array-like): A 4-element iterable [qx, qy, qz, qw]
            representing the orientation in scalar-last format.

    Returns:
        numpy.ndarray: A 4x4 homogeneous transformation matrix.
    """
    from scipy.spatial.transform import Rotation as R
    import numpy as np

    # Create rotation object from quaternion (scalar-last format)
    r = R.from_quat(quaternion)

    # Convert to 3x3 rotation matrix
    rotation_matrix = r.as_matrix()

    # Initialize 4x4 identity matrix
    transformation_matrix = np.eye(4)

    # Insert rotation
    transformation_matrix[:3, :3] = rotation_matrix

    # Insert translation
    transformation_matrix[:3, 3] = position

    return transformation_matrix

def get_all_cameras_connected() -> dict:
    """
    Retrieve all connected video camera devices available on the system.

    This function uses `cv2_enumerate_cameras` to detect available
    camera devices and returns structured information about each device,
    including its index, name, and backend driver.

    Returns:
        dict: A dictionary containing a list of detected cameras in the format:
            {
                "cameras": [
                    {
                        "index": int,
                        "name": str,
                        "backend": str
                    },
                    ...
                ]
            }

        If no cameras are detected, an empty list is returned under "cameras".
    """
    from cv2_enumerate_cameras import enumerate_cameras

    camera_list = enumerate_cameras()

    camera_list_dict = {
        "cameras": []
    }

    if camera_list:
        for i, camera_info in enumerate(camera_list):
            camera_list_dict["cameras"].append({
                "index": camera_info.index,
                "name": camera_info.name,
                "backend": camera_info.backend
            })

    return camera_list_dict

def get_all_microphones_connected() -> dict:
    """
    Retrieve all connected audio input (microphone) devices.

    This function uses the PyAudio backend to enumerate available
    audio devices and filters those capable of input (microphones).

    Returns:
        dict: A dictionary containing detected microphones in the format:
            {
                "microphones": [
                    {
                        "name": str,
                        "index": int
                    },
                    ...
                ]
            }

        Only devices with `maxInputChannels > 0` are included.
    """
    import pyaudio

    p = pyaudio.PyAudio()
    devices = p.get_device_count()

    devices_dict = {
        "microphones": []
    }

    for i in range(devices):
        device_info = p.get_device_info_by_index(i)
        if device_info.get('maxInputChannels') > 0:
            devices_dict["microphones"].append({
                "name": device_info.get('name'),
                "index": device_info.get('index')
            })

    p.terminate()

    return devices_dict

def get_all_connected_usb_interfaces() -> dict:
    """
    Retrieve all connected USB serial interfaces.

    This function uses `pyserial` to enumerate available serial ports
    and filters devices that contain "USB" in their hardware identifier.

    Returns:
        dict: A dictionary containing detected USB serial interfaces:
            {
                "usb": [
                    {
                        "name": str,
                        "Description": str,
                        "Manufacturer": str | None,
                        "Hardware ID": str
                    },
                    ...
                ]
            }

        If no USB serial devices are found, an empty list is returned.
    """
    import serial.tools.list_ports

    ports = serial.tools.list_ports.comports()

    ports_dict = {
        "usb": []
    }

    if ports:
        for port in ports:
            if "USB" in port.hwid:
                ports_dict["usb"].append(
                    {
                        "name": port.name,
                        "Description": port.description,
                        "Manufacturer": port.manufacturer,
                        "Hardware ID": port.hwid
                    }
                )
    
    return ports_dict

def get_all_ethernet_interfaces() -> dict:
    """
    Retrieve all available network (Ethernet) interfaces and their IP addresses.

    This function uses the `ifaddr` library to enumerate network adapters
    and collects their associated IP addresses and subnet prefixes.

    Returns:
        dict: A dictionary structured as:
            {
                "adapters": [
                    {
                        "name": str,
                        "Description": str,
                        "ips": [
                            "ip_address/network_prefix",
                            ...
                        ]
                    },
                    ...
                ]
            }

        Includes both IPv4 and IPv6 addresses if available.
    """
    import ifaddr

    adapters = ifaddr.get_adapters()

    adapters_dict = {
        "adapters": []
    }

    for adapter in adapters:
        adp = {}
        adp["name"] = adapter.nice_name
        adp["Description"] = adapter.name
        adp["ips"] = []

        for ip in adapter.ips:
            adp["ips"].append(f"{ip.ip}/{ip.network_prefix}")

        adapters_dict["adapters"].append(adp)
    
    return adapters_dict

def get_all_shared_memory_ressources_available(prefix: str | None = None):
    """
    List all shared memory objects available on the system (Linux only).

    On Linux, POSIX shared memory objects are accessible via `/dev/shm`.
    On non-Linux systems (e.g., Windows, macOS), this function will
    return an empty list because shared memory objects are not exposed
    through a filesystem.

    Args:
        prefix (str | None): Optional name prefix filter.

    Returns:
        list[str]: List of shared memory object names.
    """
    import os
    import sys

    # Only supported on Linux
    if sys.platform != "linux":
        return []

    shm_path = "/dev/shm"

    if not os.path.exists(shm_path):
        return []

    resources = []

    for name in os.listdir(shm_path):
        if prefix is None or name.startswith(prefix):
            resources.append(name)

    return resources
