class IKSolver():
    """
    Inverse Kinematic Solver. Uses mink library has a backbone.
    Given a end effector position generates a collision-free trajectory of qpos.
    """
    def __init__(
            self,
            model_path,
            attachement_frame_name : str,
            frame_type : AttachementFrameType,
            initial_configuration : list[float],
            orientation_cost: float = 1.0,
            position_cost: float = 1.0,
            lm_damping: float = 1e-3,
            posture_task: mink.PostureTask | None = None,
            velocity_limit: mink.VelocityLimit | None = None,
            collision_limit: mink.CollisionAvoidanceLimit | None = None,
            solver : str = "daqp",
            steps : int = 1000,
            gain : float = 0.5,
            dt : float = 0.01,
        ):
        """
        Args:
            model_path: path to your robot xml mujoco representation.
            attattachement_frame_name: end effector frame name inside your xml mujoco model.
            frame_type: type of your end effector frame inside your xml mujoco model.
            initial_configuration: starting qpos configuration.
            solver: Inverse kinematics solver type. default to 'daqp'.
            gain:
            steps: number of step to run until end of convergence
            dt: 
            orientation_cost: end effector orientation cost
            position_cost: end effetor position cost
            lm_damping: 
            posture_task: Posture limitation task
            velocity_limit: Velocity limitation
            collision_limit: Collision Avoidance limitation
        """
        self.model_path_ = model_path
        self.attachement_frame_ = attachement_frame_name
        self.frame_type_ = frame_type

        self.initial_configuration_ = initial_configuration
        self.orientation_cost_ = orientation_cost
        self.position_cost_ = position_cost
        self.lm_damping_ = lm_damping

        self.solver_ = solver

        self.steps_ = steps
        self.gain_ = gain
        self.dt_ = dt

        # optionnal configuration
        self.posture_task_ = posture_task
        self.velocity_limit_ = velocity_limit
        self.collision_limit_ = collision_limit

        self.configuration_ = None
        self.model_ = None
        self.limits_ = []
        self.tasks_ = []

        self.checking_intialization()

        self._initiliaze_configuration()

    def _initiliaze_configuration(self) -> None:
        """
        Private method called at the end of the constructor.
        Intialize IK solver and end effector.
        """
        self.model_ = mujoco.MjModel.from_xml_path(self.model_path_)
        self.configuration_ = mink.Configuration(self.model_)

        self.ee_task_ = mink.FrameTask(
            frame_name=self.attachement_frame_,
            frame_type=self._get_frame_type(),
            position_cost=self.position_cost_,
            orientation_cost=self.orientation_cost_,
            lm_damping=self.lm_damping_,
        )

        self.limits_.append(
            mink.ConfigurationLimit(self.model_)
        )

        self.tasks_ = [self.ee_task_]

        self.configuration_.update(q=self.initial_configuration_)

        # optional configuration intialization
        if self.posture_task_ is not None:
            self.tasks_.append(self.posture_task_)

        if self.collision_limit_ is not None:
            self.limits_.append(self.collision_limit_)

        if self.velocity_limit_ is not None:
            self.limits_.append(self.velocity_limit_)

    def create_collision_limit(
            self, 
            collision_pairs: dict,
            minimum_distance_from_collisions:float = 0.03,
            collision_detection_distance: float = 0.01
    ) -> mink.CollisionAvoidanceLimit:
        """
        Create a Collision avoidance limitation object

        Args:
            collision_pairs: dict of joint to avoid collision with.
            minimum_distance_from_collisions: how far should I distance myself from a collision
            collision_detection_distance: of far should I start detecting has a collision

        Returns:
            A mink Collision avoidance limit object.
        """
        return mink.CollisionAvoidanceLimit(
                    self.model_,
                    geom_pairs=collision_pairs,
                    minimum_distance_from_collisions=minimum_distance_from_collisions,
                    collision_detection_distance=collision_detection_distance,
                    gain=self.gain_,
                )

    def solveIK(self, T_target) -> list[list[float]]:
        """
        Compute of collision-free trajectory of qpos.

        Args:
            T_target: pose of the goal position

        Returns:
            IK informations.
        """
        self.ee_task_.set_target(T_target)


        trajectory = self._IK_loop(
            T_target=T_target,
        )

        # Check result.
        final = self.get_current_eef_pose()
        error = np.linalg.norm(final.translation() - T_target.translation())
        print(f"Position error: {error} m")

        return {
            "trajectory": trajectory,
            "final_error": error,
            "converged": error < 1e-3,
            "steps": len(trajectory)
        }




    def _get_frame_type(self):
        """
        Private helper method. Gets the type of end effector attachement as a string.
        """
        if self.frame_type_ == AttachementFrameType.SITE: return "site"
        elif self.frame_type_ == AttachementFrameType.GEOM: return "geom"

    def solveIK_viewer(self, T_target: mink.SE3):
        """
        Same as solveIK, but visualizes the IK solution in a MuJoCo viewer.
        Compute of collision-free trajectory of qpos.

        Args:
            T_target: pose of the goal position

        Returns:
            IK informations.
        """

        self.ee_task_.set_target(T_target)

        # Launch viewer
        with mujoco.viewer.launch_passive(
            model=self.model_,
            data=self.configuration_.data,
            show_left_ui=False,
            show_right_ui=False
        ) as viewer:
            # Set camera to a reasonable initial view
            mujoco.mjv_defaultFreeCamera(self.model_, viewer.cam)
            rate = RateLimiter(frequency=200.0, warn=False)
            
            trajectory = self._IK_loop(
                T_target=T_target,
                viewer=viewer,
                rate=rate
            )
            
            while True:
                rate.sleep()
                response = input("press x to finish rendering...")
                if response == "x":
                    break

            # Check result.
            final = self.get_current_eef_pose()
            error = np.linalg.norm(final.translation() - T_target.translation())
            print(f"Position error: {error} m")

        return {
            "trajectory": trajectory,
            "final_error": error,
            "converged": error < 1e-3,
            "steps": len(trajectory)
        }


    

    def _IK_loop(self, T_target, viewer=None, rate=None):
        """
        Private method, that performs thew IK loop.

        Args:
            T_target: target pose.
            viewer: mujoco vierwer (default: None)
            rate: update viewer rate (default: None)
        
        Returns:
            trajectory of qpos.
        """
        trajectory = []
        for _ in range(self.steps_):
            # Solve IK velocity
            dq = mink.solve_ik(
                self.configuration_,
                self.tasks_,
                self.dt_,
                solver=self.solver_,
                limits=self.limits_ if len(self.limits_) >= 1 else None,
            )

            # damping/gain
            dq *= self.gain_

            # Integrate
            self.configuration_.integrate_inplace(dq, self.dt_)
            trajectory.append(self.configuration_.q.copy())

            # Required collision/camera updates
            mujoco.mj_fwdPosition(self.model_, self.configuration_.data)
            mujoco.mj_sensorPos(self.model_, self.configuration_.data)
            mujoco.mj_camlight(self.model_, self.configuration_.data)

            # Sync viewer
            if viewer != None:
                viewer.sync()
                
            if rate != None:
                rate.sleep()

            # check convergence
            final = self.get_current_eef_pose()
            error = np.linalg.norm(final.translation() - T_target.translation())
            if error < 1e-3:
                print(f"error : {error}")
                break
        
        return trajectory
