class Renderer():
    """
    Mujoco Renderer.
    Gives a visual preview of a mujoco state.
    """
    def __init__(
            self,
            model_path : str,
            width : int,
            height : int,
            debug: bool = False
        ):
        """
        Args:
            model_path: path to your robot xml mujoco representation.
            width: width of the video/image preview
            height: height of the video/image preview
        """
        self.model_path_ = model_path
        self.width = width
        self.height = height
        self._debug = debug

        self.model = None

        # streaming
        self._streaming = False
        self._stream_thread = None
        self.frame = None
        self._shared_memory_server = None
        self._shared_memory_array = None
        self._frame_shape = (self.height, self.width, 3)
        self._buffer_size = int(np.prod(self._frame_shape)) # 921600

        self.renderer = self._initialize_renderer()

    def debug(self, msg: str) -> None:
        if self._debug:
            print(f"[DEBUG RENDERER] {msg}")



    def _initialize_renderer(self) -> None:
        """
        Private method called at the end of the constructor.
        Initiliaze the mujoco renderer.
        """
        self.model = mujoco.MjModel.from_xml_path(self.model_path_)
        self.data = mujoco.MjData(self.model)

        self.camera = mujoco.MjvCamera()
        mujoco.mjv_defaultFreeCamera(self.model, self.camera)

        self.camera.lookat[:] = [0.0, 0.0, 0.0]   # center of robot
        self.camera.distance = 1.5

        renderer = self._create_renderer_object()

        self.debug("renderer initialized")
        return renderer

    def _create_renderer_object(self) -> mujoco.Renderer:
        return mujoco.Renderer(
            model=self.model,
            height= self.height,
            width= self.width
        )

    def enable_shared_memory_stream(self, name: str = "orionStream") -> None:
        """
        Create a shared memory server for image streaming.
        """
        self._shared_memory_server = shared_memory.SharedMemory( # this should be ok
            name=name,
            create=True,
            size=self._buffer_size  # 921600
        )

        self._shared_memory_array = np.ndarray(     # this should be ok
            self._frame_shape,
            dtype=np.uint8,
            buffer=self._shared_memory_server.buf
        )

        self.debug(f"shared memory enabled at {name}")

    def take_screen_shot(self) -> str:
        """
        Generate a image of the current mujoco state.

        Returns:
            Generated image path.
        """
        mujoco.mj_forward(self.model, self.data)
        self.renderer.update_scene(self.data, self.camera)
        
        pixels = self.renderer.render()
        name = uuid.uuid4()
        image_path = f'{BASE_PATH}/{name}.png'

        image = Image.fromarray(pixels)
        image.save(image_path)

        return image_path

    def start_realtime_stream(self, fps: int = 30) -> None:
        """
        Start real-time rendering in a background thread.

        Args:
            fps: target frames per second
        """
        if self._streaming:
            return

        self._streaming = True
        period = 1.0 / fps

        def loop():
            thread_renderer = self._create_renderer_object()
            while self._streaming:
                start_time = time.time()

                mujoco.mj_forward(self.model, self.data)
                thread_renderer.update_scene(self.data, self.camera)
                frame = thread_renderer.render()
                self.frame = frame

                self.publish_to_shared_memory()

                elapsed = time.time() - start_time
                sleep_time = max(0, period - elapsed)
                time.sleep(sleep_time)

        self._stream_thread = threading.Thread(target=loop, daemon=True)
        self._stream_thread.start()

    def publish_to_shared_memory(self) -> None:
        """ 
        Write a numpy image to shared memroy. For easy and realtime acess between process.
        """
        if self._shared_memory_array is not None:
            self._shared_memory_array[:] = self.frame

    def close_shared_memory(self) -> None:
        """    
        Close shared memory.
        """
        if self._shared_memory_server is not None:
            self._shared_memory_server.close()
            self._shared_memory_server.unlink()
            self._shared_memory_server = None

    def stop_realtime_stream(self) -> None:
        """
        Stop real-time rendering thread.
        """
        if not self._streaming:
            return

        self._streaming = False

        if self._stream_thread is not None:
            self._stream_thread.join()
            self._stream_thread = None
